63 bool Setup::load_from_file(const char *setup_filename) {
64  bool config_changed = false;
65  char config_buffer[SETUP_FILE_PARSE_BUFFER_SIZE];
66  char *key = config_buffer;
67  char *value = config_buffer;
68 
69  // open the setup file
70  auto setup_file = SD.open(setup_filename, FILE_READ);
71 
72  if (!setup_file)
73  return false;
74 
75  // Process each config file lines
76  char c;
77  uint32_t pos = 0;
78  LineParseState pstate = LINE_START;
79  bool parsed_line = false;
80 
81  while (true) {
82  c = (char)setup_file.read();
83  if ((int8_t)c == -1)
84  break;
85 
86  switch (pstate) {
87  case LINE_START:
88  if (c != ' ' && c != '\n' && c != '\r') {
89  // found beginning of key
90  config_buffer[pos] = c;
91  pos++; // pos == 1 after this line
92  pstate = LINE_PARSE_KEY;
93  }
94  break;
95 
96  case LINE_PARSE_KEY:
97  if (pos == SETUP_FILE_PARSE_BUFFER_SIZE) {
98  pstate = LINE_SKIP;
99  } else if (c == ' ') {
100  // found end of the key, mark and go to the value
101  config_buffer[pos] = '\0'; // mark the end of the string
102  pos++;
103  pstate = LINE_PARSE_DELIM;
104  } else if (c == '=') {
105  config_buffer[pos] = '\0';
106  pos++;
107  pstate = LINE_PARSE_DELIM;
108  } else if (c == '\n' || c == '\r') {
109  // premature end of line, skip
110  pos = 0;
111  pstate = LINE_START;
112  } else {
113  config_buffer[pos] = c;
114  pos++;
115  }
116  break;
117 
118  case LINE_PARSE_DELIM:
119  if (pos == SETUP_FILE_PARSE_BUFFER_SIZE) {
120  pstate = LINE_SKIP;
121  } else if (c == '\n' || c == '\r') {
122  // we also accept an empty value
123  value = config_buffer + pos;
124  config_buffer[pos] = '\0';
125  pos++;
126  parsed_line = true; // mark parsing as successful
127  // parse next line
128  pos = 0;
129  pstate = LINE_START;
130  } else if (c != ' ' && c != '=') {
131  value = config_buffer + pos;
132  config_buffer[pos] = c;
133  pos++;
134  pstate = LINE_PARSE_VALUE;
135  }
136  break;
137 
138  case LINE_PARSE_VALUE:
139  if (pos == SETUP_FILE_PARSE_BUFFER_SIZE) {
140  pstate = LINE_SKIP;
141  } else if (c == '\n' || c == '\r' || c == ' ') {
142  // end of value
143  config_buffer[pos] = '\0';
144  parsed_line = true; // indicate successful parsing
145  pos = 0;
146  pstate = LINE_START;
147  } else {
148  config_buffer[pos] = c;
149  pos++;
150  }
151  break;
152 
153  case LINE_SKIP:
154  // wait until we find the end of line
155  if (c == '\n' || c == '\r') {
156  pos = 0;
157  pstate = LINE_START;
158  }
159  break;
160  }
161 
162  if (parsed_line) {
163  parsed_line = false;
164 
165  //
166  // Process matching keys
167  //
168  if (std::strcmp(key, "cpmf") == 0) {
169  // Update cpm factor
170  float factor = std::atof(value);
171  if (_config.cpm2ush_divider != factor && factor > 0.0) {
172  _config.cpm2ush_divider = factor;
173  config_changed = true;
174  }
175  } else if (std::strcmp(key, "bqmf") == 0) {
176  // Update bq/m2 factor
177  float factor = std::atof(value);
178  if (_config.cpm2bqm2_factor != factor && factor > 0.0) {
179  _config.cpm2bqm2_factor = factor;
180  config_changed = true;
181  }
182  } else if (std::strcmp(key, "nm") == 0) {
183  // Update name in EEPROM
184  if (std::strcmp(_config.user_name, value) != 0) {
185  std::strncpy(_config.user_name, value, SETUP_USERNAME_MAXLEN);
186  config_changed = true;
187  }
188  } else if (std::strcmp(key, "did") == 0) {
189  // Update device id in persistent storage
190  if (_config.device_id != (unsigned int)std::atoi(value)) {
191  _config.device_id = std::atoi(value);
192  config_changed = true;
193  }
194  } else if (std::strcmp(key, "gt") == 0) {
195  // Update geiger type in persistent storage
196  if (_config.type != std::atoi(value)) {
197  _config.type = std::atoi(value);
198  config_changed = true;
199  }
200  } else if (std::strcmp(key, "gm") == 0) {
201  // Update geiger mode in persistent storage
202  if (_config.mode != std::atoi(value)) {
203  _config.mode = std::atoi(value);
204  config_changed = true;
205  }
206  } else if (std::strcmp(key, "al") == 0) {
207  // Update alarm threshold in persistent storage
208  uint32_t new_alarm_level = std::atoi(value);
209  if (_config.alarm_level != new_alarm_level) {
210  _config.alarm_level = new_alarm_level;
211  config_changed = true;
212  }
213  } else if (std::strcmp(key, "cn") == 0) {
214  // Update country code in persistent storage
215  if (std::strcmp(_config.country_code, value) != 0) {
216  std::strncpy(_config.country_code, value,
217  sizeof(SETUP_DEFAULT_COUNTRY_CODE) - 1);
218  config_changed = true;
219  }
220  } else if (std::strcmp(key, "tz") == 0) {
221  // Update timezone in persistent storage
222  uint8_t new_timezone = std::atoi(value);
223  if (_config.timezone != new_timezone) {
224  _config.timezone = new_timezone;
225  config_changed = true;
226  }
227  } else if (std::strcmp(key, "st") == 0) {
228  // Update sensor type in persistent storage
229  uint8_t new_type = std::atoi(value);
230  if (_config.sensor_type != new_type) {
231  _config.sensor_type = new_type;
232  config_changed = true;
233  }
234  } else if (std::strcmp(key, "ss") == 0) {
235  // Update sensor shield in persistent storage
236  uint8_t new_sensor_shield = std::atoi(value);
237  if (_config.sensor_shield != new_sensor_shield) {
238  _config.sensor_shield = new_sensor_shield;
239  config_changed = true;
240  }
241  } else if (std::strcmp(key, "sh") == 0) {
242  // Update sensor height in persistent storage
243  uint32_t new_sensor_height = std::atoi(value);
244  if (_config.sensor_height != new_sensor_height) {
245  _config.sensor_height = new_sensor_height;
246  config_changed = true;
247  }
248  } else if (std::strcmp(key, "sm") == 0) {
249  // Update sensor mode in persistent storage
250  if (_config.sensor_mode != std::atoi(value)) {
251  _config.sensor_mode = std::atoi(value);
252  config_changed = true;
253  }
254  } else if (std::strcmp(key, "dose") == 0) {
255  // Reset total dose in persistent storage
256  std::memset(&_dose, 0, sizeof(DoseData));
257  _prefs.putBytes(SETUP_KEY_DOSE, &_dose, sizeof(DoseData));
258  }
259  }
260  }
261 
262  // close the setup file
263  setup_file.close();
264 
265  if (config_changed) {
266  // Configuration is changed
267  _prefs.putBytes(SETUP_KEY_CONFIG, &_config, sizeof(ConfigData));
268  }
269 
270  return true;
271 }
