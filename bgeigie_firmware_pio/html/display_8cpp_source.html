27void Display::update() {
28 // This function is mainly used to read the buttons and update the state
29 // of the display
30
31 bool button_A_pressed = M5.BtnA.wasPressed();
32 bool button_B_pressed = M5.BtnB.wasPressed();
33 bool button_C_pressed = M5.BtnC.wasPressed();
34
35 switch (state) {
36 case (S_STARTUP):
37 clear(); // initialize display
38 draw_base();
39 state = S_MAIN_DRAW; // next state
40 break;
41
42 case (S_MAIN_DRAW):
43 clear();
44 draw_base();
45 state = S_MAIN_SHOW;
46 case (S_MAIN_SHOW):
47 draw_main();
48 if (button_A_pressed) state = S_QRCODE_DRAW;
49 break;
50
51 case (S_QRCODE_DRAW):
52 clear();
53 draw_qrcode();
54 state = S_QRCODE_SHOW;
55 break;
56
57 case (S_QRCODE_SHOW):
58 if (button_A_pressed) state = S_MAIN_DRAW;
59 break;
60 }
61}

171void printFloat(float val, bool valid, int len, int prec) {
172 if (!valid) {
173 while (len-- > 1) M5.Lcd.print('*');

176 M5.Lcd.print(val, prec);
