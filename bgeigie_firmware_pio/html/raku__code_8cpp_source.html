431  if(value < 0){

438 void ip5306_set_bits(uint8_t reg, uint8_t index, uint8_t bits, uint8_t value){
439  uint8_t mask = (1 << bits) - 1;

441  if(v < 0){
442  Serial.printf("ip5306_get_reg fail: 0x%02x\n", reg);
443  return;
444  }
445  v &= ~(mask << index);
446  v |= ((value & mask) << index);
447  if(ip5306_set_reg(reg, v)){
448  Serial.printf("ip5306_set_bits fail: 0x%02x\n", reg);
