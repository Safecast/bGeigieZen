250bool LocalStorage::activate(bool) {
251 _memory.begin(memory_name, true);
252 _device_id = _memory.getUShort(key_device_id, D_DEVICE_ID);
253 if(_memory.getString(key_ap_password, _ap_password, CONFIG_VAL_MAX) == 0) {
254 strcpy(_ap_password, D_ACCESS_POINT_PASSWORD);
255 }
256 if(_memory.getString(key_wifi_ssid, _wifi_ssid, CONFIG_VAL_MAX) == 0) {
257 strcpy(_wifi_ssid, D_WIFI_SSID);
258 }
259 if(_memory.getString(key_wifi_password, _wifi_password, CONFIG_VAL_MAX) == 0) {
260 strcpy(_wifi_password, D_WIFI_PASSWORD);
261 }
262 if(_memory.getString(key_api_key, _api_key, CONFIG_VAL_MAX) == 0) {
263 strcpy(_api_key, D_APIKEY);
264 }
265 _fixed_longitude = _memory.getDouble(key_fixed_longitude, 0);
266 _fixed_latitude = _memory.getDouble(key_fixed_latitude, 0);
267 _fixed_range = _memory.getUInt(key_fixed_range, 0);
268 _last_longitude = _memory.getDouble(key_last_longitude, 0);
269 _last_latitude = _memory.getDouble(key_last_latitude, 0);
270 _memory.end();
271 return true;
272}
