180bool BluetoothReporter::send(const DataLine& reading) const {
181 if(_pServer->getConnectedCount() == 0) {
182 // No clients to send data to
183 return false;
184 }
185 DEBUG_PRINTLN("Sending reading over Bluetooth");
186 size_t size = strlen(reading.log_string);
187
188 int segment = 0;
189 const static uint8_t max_segment_size = 20; // Max that can be sent over bluetooth
190 do {
191 ++segment;
192 uint8_t segment_size = segment * max_segment_size > size ? size % max_segment_size : max_segment_size;
193 char to_send[segment_size];
194 strncpy(to_send, reading.log_string + ((segment - 1) * max_segment_size), segment_size);
195
196 pDataRXCharacteristic->setValue((uint8_t*) to_send, segment_size);
197 pDataRXCharacteristic->notify();
198 } while(segment * max_segment_size < size);
199 return true;
200}
