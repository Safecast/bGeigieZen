123
124 if (!setup_file) {
125 return SdStatus::e_sd_config_status_config_no_content;
126 }
127
128 uint32_t file_device_id = 0;
129
130 while (setup_file.available()) {
131 String line = setup_file.readStringUntil('\n');
132 if (line.startsWith(SD_CONFIG_FIELD_DEVICE_ID)) {
133 sscanf(line.c_str(), sd_config_device_id_f, &file_device_id);
134 break;
135 }
136 }
137
138 // close the setup file
139 setup_file.close();
140
141 if (file_device_id == 0) {
142 return SdStatus::e_sd_config_status_config_no_content;
143 } else if (device_id && file_device_id != device_id) {
144 return SdStatus::e_sd_config_status_config_different_id;
145 } else {
146 return SdStatus::e_sd_config_status_ok;
147 }
