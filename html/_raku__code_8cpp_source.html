56//Compute CPM
57 unsigned long cpm_gen()
58 {
59 unsigned int i;
60 unsigned long c_p_m = 0;
61
62 // sum up
63 for (i = 0 ; i < NX ; i++)
64 c_p_m += shift_reg[i];
65 //deadtime compensation (medcom international)
66 c_p_m = (unsigned long)((float)c_p_m / (1 - (((float)c_p_m * 1.8833e-6))));
67 return c_p_m;
68 }

86int interruptCounterAvailable()
87{
88 // get current time
89 unsigned long now = millis();
90 // do basic check for millis overflow
91 if (now >= _start_time)
92 return (now - _start_time >= _delay);
93 else
94 return (ULONG_MAX + now - _start_time >= _delay);
95}

106 {
107 Wire.beginTransmission(0x75);

111 switch (Wire.read() & 0xF0) {
112 case 0xE0: return 25;
113 case 0xC0: return 50;

385uint8_t ip5306_get_bits(uint8_t reg, uint8_t index, uint8_t bits){
386 int value = ip5306_get_reg(reg);
387 if(value < 0){
388 Serial.printf("ip5306_get_bits fail: 0x%02x\n", reg);
389 return 0;
390 }
391 return (value >> index) & ((1 << bits)-1);
392}
393
394void ip5306_set_bits(uint8_t reg, uint8_t index, uint8_t bits, uint8_t value){
395 uint8_t mask = (1 << bits) - 1;
396 int v = ip5306_get_reg(reg);
397 if(v < 0){
398 Serial.printf("ip5306_get_reg fail: 0x%02x\n", reg);
399 return;
400 }
401 v &= ~(mask << index);
402 v |= ((value & mask) << index);
403 if(ip5306_set_reg(reg, v)){
404 Serial.printf("ip5306_set_bits fail: 0x%02x\n", reg);
405 }
406}

486{

511 M5.Lcd.println();

513 printTime(gps.time);
514 M5.Lcd.println();
515 M5.Lcd.print("Altitude :");
516 printFloat(gps.altitude.meters(), gps.altitude.isValid(), 7, 2);
517 M5.Lcd.println();
518 M5.Lcd.print("Degree :");
519 printFloat(gps.course.deg(), gps.course.isValid(), 7, 2);
520 M5.Lcd.println();
521 M5.Lcd.print("Speed :");
522 printFloat(gps.speed.kmph(), gps.speed.isValid(), 6, 2);
523
524 M5.Lcd.println();
525
526 smartDelay(1000);
527
528 // send IP5003 data to serial
529 printIP5306Stats();
530
531 if (millis() > 5000 && gps.charsProcessed() < 10)
532 M5.Lcd.println(F("No GPS data received: check wiring"));
533
534 // Send reading data if IS_READY is passed when
535 if IS_READY {
536
537 cps = interruptCounterCount();
538 interruptCounterReset();
539
540 // insert count in sliding window and compute CPM
541 shift_reg[reg_index] = cps; // put the count in the correct bin
542 reg_index = (reg_index + 1) % NX; // increment register index
543 cpm = cpm_gen(); // compute sum over all bins
544
545
546 // Convert CPM to uSvH
547 conversionCoefficient = 1 / sensor1_cpm_factor;
548 float uSv = cpm * conversionCoefficient; // convert CPM to Micro Sievers Per Hour
549 char CPM_string[16];
550 dtostrf(cpm, 0, 0, CPM_string);
551 if (cpm > (int)(S1peak)) {
552 S1peak = cpm;
553 }
554
555 M5.Lcd.setTextColor(TFT_WHITE, TFT_BLACK);
556 //Display CPM
557 M5.Lcd.setCursor(20,50);
558 M5.Lcd.drawString("CPM =",5, 50, 4);
559 M5.Lcd.setCursor(120,55);
560 // printInt(cpm, true, 5);
561 printIntFont(cpm, true, 5, 50, 90, 4);
562
563 //Display uSv/h
564 M5.Lcd.setCursor(22,70);
565 M5.Lcd.drawString("uSv/h =",5, 70, 4);
566 M5.Lcd.setCursor(100,80);
