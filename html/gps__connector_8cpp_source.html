60 // Assume that the U-Blox GNSS is running at 9600 baud (the default) or at 38400 baud.
61 if (!retry) {

68 _init_at = millis();

70 if (_tried_115200_at == 0 && (millis() - _init_at > 1000)) { // Wait for device to completely startup
71 _tried_115200_at = millis();
72 _serial_conn.updateBaudRate(115200);
73 M5_LOGD("GNSS: Try at 115200 baud");
74 if (_gnss.begin(_serial_conn, 500)) {
75 M5_LOGD("GNSS: connected at 115200 baud"); // no need to set module to 38.4

81 else if (_tried_38400_at == 0 && _tried_115200_at > 0 && (millis() - _tried_115200_at > 1200)) { // Wait for device to completely startup
82 _tried_38400_at = millis();
83 _serial_conn.updateBaudRate(38400);
84 M5_LOGD("GNSS: Try at 38400 baud");
85 if (_gnss.begin(_serial_conn, 500)) {
86 M5_LOGD("GNSS: connected at 38400 baud"); // no need to set module to 38.4

91 }
92 else if (_tried_9600_at == 0 && _tried_38400_at > 0 && (millis() - _tried_38400_at > 1200)) {
93 _tried_9600_at = millis();
94 _serial_conn.updateBaudRate(9600);
95 M5_LOGD("GNSS: Try at 9600 baud");
96 if (_gnss.begin(_serial_conn, 500)) {
97 M5_LOGD("GNSS: connected at 9600 baud, switching to 38400");
98 _gnss.setSerialRate(38400);
99 delay(100); // recovery time for gnss module baud rate change
100 _serial_conn.updateBaudRate(38400);
101 }
102 else {
103 return false;
104 }
105 }
106 else {
107 return false;
108 }
109
110 // Confirm that we actually have a connection
111 data.protocolVersionHigh = _gnss.getProtocolVersionHigh();
112 data.protocolVersionLow = _gnss.getProtocolVersionLow();
113 M5_LOGD("GNSS: u-blox protocol version %02d.%02d",
114 data.protocolVersionHigh, data.protocolVersionLow);
115
116 // Send UBX, disable NMEA-0183 messages that we are ignoring anyway.
117 _gnss.setPortOutput(COM_PORT_UART1, COM_TYPE_UBX);
118
119 // Set Auto on NAV-PVT for non-blocking access
120 // getPVT() will return true if a new navigation solution is available
121 _gnss.setAutoPVT(true); // Tell the GNSS to send the solution as it is computed (1 second)
122 _gnss.setAutoNAVSAT(false); // Disable navsat by default (navsat worker handles this)
123

137 // getPVT returns true if there is a fresh navigation solution available.
138 // "LLH" is longitude, latitude, height.
139 // getPVT() returns UTC date and time.
140 // Do not use GNSS time, see u-blox spec section 9.
141 if (_gnss.getPVT()) {
142 // M5_LOGD("[%d] _gnss.getPVT() is true.", millis());
143
144 data.satsInView = _gnss.getSIV(); // Satellites In View
145
146 if (_gnss.getFixType() == 2 || _gnss.getFixType() == 3) {
147 // M5_LOGD("[%d] fix type is 2D or 3D.", millis());
148 data.pdop = _gnss.getPDOP() * 1e-2; // Position Dilution of Precision
149 _last_latitude = data.latitude;
150 _last_longitude = data.longitude;
151 data.latitude = _gnss.getLatitude() * 1e-7;
152 data.longitude = _gnss.getLongitude() * 1e-7;
153 data.heading_degree = _gnss.getHeading() * 1e-5;
154 if (data.heading_degree >= NNW || data.heading_degree < NNE) {
155 data.heading = GnssData::NORTH;
156 }
157 if (data.heading_degree >= NNE || data.heading_degree < ENE) {
158 data.heading = GnssData::NORTHEAST;
159 }
160 if (data.heading_degree >= ENE || data.heading_degree < ESE) {
161 data.heading = GnssData::EAST;
162 }
163 if (data.heading_degree >= ESE || data.heading_degree < SSE) {
164 data.heading = GnssData::SOUTHEAST;
165 }
166 if (data.heading_degree >= SSE || data.heading_degree < SSW) {
167 data.heading = GnssData::SOUTH;
168 }
169 if (data.heading_degree >= SSW || data.heading_degree < WSW) {
170 data.heading = GnssData::SOUTHWEST;
171 }
172 if (data.heading_degree >= WSW || data.heading_degree < WNW) {
173 data.heading = GnssData::WEST;
174 }
175 if (data.heading_degree >= WNW || data.heading_degree < NNW) {
176 data.heading = GnssData::NORTHWEST;
177 }
178 data.altitudeMSL = _gnss.getFixType() == 3 ? _gnss.getAltitudeMSL() * 1e-3 : 0; // Above MSL (not ellipsoid)
179 const auto distance_step = haversine_km(data.latitude, data.longitude, _last_latitude, _last_longitude);
180 data.location_valid = _gnss.getGnssFixOk() && distance_step < 0.5; // Airplanes fly at 255m per second, 500 meter check is good enough
181 location_timer.restart();
182 ret_status = e_worker_data_read;
183 // DEBUG: Compare PDOP from NAV-PVT and HDOP from NAV-DOP
184 // M5_LOGD("[%d] GnssFixOk (type = %d)."
185 // " SATS: %d; PDOP: %d; HDOP: %d\n",
186 // millis(), _gnss.getFixType(),
187 // data.satsInView, _gnss.getPDOP(), data.hdop);

190 if (_gnss.getDateValid()) {

192 data.year = _gnss.getYear();
193 data.month = _gnss.getMonth();
194 data.day = _gnss.getDay();
195 data.date_valid = true;

197 ret_status = e_worker_data_read;
198 }
199
200 if (_gnss.getTimeValid()) {
201// M5_LOGD("[%lu] _gnss.getTimeValid() is true.", millis());
202 data.hour = _gnss.getHour();
203 data.minute = _gnss.getMinute();
204 data.second = _gnss.getSecond();
205 data.time_valid = true;
206 time_timer.restart();
207 ret_status = e_worker_data_read;
208 }
209 }
210
211 // Check expiry
212 if (location_timer.isExpired()) {
213 data.location_valid = false;
214 }
215 if (time_timer.isExpired()) {
216 data.time_valid = false;
217 data.hour = GPS_INVALID_HOUR;
218 data.minute = GPS_INVALID_MINUTE;
219 data.second = GPS_INVALID_SECOND;
220 }
221 if (date_timer.isExpired()) {
222 data.date_valid = false;
223 data.year = GPS_INVALID_YEAR;
224 data.month = GPS_INVALID_MONTH;
225 data.day = GPS_INVALID_DAY;
226 }
227
