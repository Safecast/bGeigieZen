176int8_t BluetoothReporter::handle_async() {
177 if (_pServer->getConnectedCount() == 0) {
178 // No clients to send data to
179 return e_bluetooth_no_clients;
180 }
181 size_t size = strlen(_log_string);
182
183 int segment = 0;
184 const static uint8_t max_segment_size = 20; // Max that can be sent over bluetooth
185 do {
186 ++segment;
187 uint8_t segment_size = segment * max_segment_size > size ? size % max_segment_size : max_segment_size;
188 char to_send[segment_size];
189 strncpy(to_send, _log_string + ((segment - 1) * max_segment_size), segment_size);
190
191 _pDataRXCharacteristic->setValue((uint8_t*) to_send, segment_size);
192 _pDataRXCharacteristic->notify();
193 } while (segment * max_segment_size < size);
194 return e_bluetooth_client_sent;
195}
