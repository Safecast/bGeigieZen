84bool TinyGPS::encode(char c)
85{
86 bool valid_sentence = false;
87
88#ifndef _GPS_NO_STATS
89 ++_encoded_characters;
90#endif
91 switch(c)
92 {
93 case ',': // term terminators
94 _parity ^= c;
95 case '\r':
96 case '\n':
97 case '*':
98 if (_term_offset < sizeof(_term))
99 {
100 _term[_term_offset] = 0;
101 valid_sentence = term_complete();
102 }
103 ++_term_number;
104 _term_offset = 0;
105 _is_checksum_term = c == '*';
106 return valid_sentence;
107
108 case '$': // sentence begin
109 _term_number = _term_offset = 0;
110 _parity = 0;
111 _sentence_type = _GPS_SENTENCE_OTHER;
112 _is_checksum_term = false;
113 _gps_data_good = false;
114 return valid_sentence;
115 }
116
117 // ordinary characters
118 if (_term_offset < sizeof(_term) - 1)
119 _term[_term_offset++] = c;
120 if (!_is_checksum_term)
121 _parity ^= c;
122
123 return valid_sentence;
124}

138int TinyGPS::from_hex(char a)
139{
140 if (a >= 'A' && a <= 'F')
141 return a - 'A' + 10;
142 else if (a >= 'a' && a <= 'f')
143 return a - 'a' + 10;
144 else
145 return a - '0';
146}

148unsigned long TinyGPS::parse_decimal()
149{
150 char *p = _term;
151 bool isneg = *p == '-';
152 if (isneg) ++p;
153 unsigned long ret = 100UL * gpsatol(p);
154 while (gpsisdigit(*p)) ++p;
155 if (*p == '.')
156 {
157 if (gpsisdigit(p[1]))
158 {
159 ret += 10 * (p[1] - '0');
160 if (gpsisdigit(p[2]))
161 ret += p[2] - '0';
162 }
163 }
164 return isneg ? -ret : ret;
165}

167unsigned long TinyGPS::parse_degrees() {
168 char *p;
169 unsigned long left = gpsatol(_term);
170 unsigned long tenk_minutes = (left % 100UL) * 100000UL;
171 for (p = _term; gpsisdigit(*p); ++p);
172 if (*p == '.')
173 {
174 unsigned long mult = 10000;
175 while (gpsisdigit(*++p))
176 {
177 tenk_minutes += mult * (*p - '0');
178 mult /= 10;
179 }
180 }
181 return (left / 100) * 1000000 + (tenk_minutes + 5) / 6;
182}

184unsigned long TinyGPS::parse_degrees_raw()
185{
186 char *p;
187 unsigned long left = gpsatol(_term);
188 unsigned long right = 0;
189 for (p=_term; gpsisdigit(*p); ++p);
190 if (*p == '.')
191 {
192 unsigned long mult = 10000;
193 while (gpsisdigit(*++p))
194 {
195 right += mult * (*p - '0');
196 mult /= 10;
197 }
198 }
199 return (left * 100000) + right;
200}

204// Processes a just-completed term
205// Returns true if new sentence has just passed checksum test and is validated
206bool TinyGPS::term_complete()
207{
208 if (_is_checksum_term)
209 {
210 uint8_t checksum = 16 * from_hex(_term[0]) + from_hex(_term[1]);
211 if (checksum == _parity)
212 {
213 if (_gps_data_good)
214 {
215#ifndef _GPS_NO_STATS
216 ++_good_sentences;
217#endif
218 _last_time_fix = _new_time_fix;
219 _last_position_fix = _new_position_fix;
220
221 switch(_sentence_type)
222 {
223 case _GPS_SENTENCE_GPRMC:
224 _time = _new_time;
225 _date = _new_date;
226 _latitude = _new_latitude;
227 _longitude = _new_longitude;
228 _speed = _new_speed;
229 _course = _new_course;
230 break;
231 case _GPS_SENTENCE_GPGGA:
232 _altitude = _new_altitude;
233 _time = _new_time;
234 _latitude = _new_latitude;
235 _longitude = _new_longitude;
236 _numsats = _new_numsats;
237 _hdop = _new_hdop;
238 break;
239 }
240
241 return true;
242 } else {
243 // Handle MTK3339 RTC clock (even with no GPS fix)
244 if (_new_time) {
245 _last_time_fix = _new_time_fix;
246 switch(_sentence_type)
247 {
248 case _GPS_SENTENCE_GPRMC:
249 _time = _new_time;
250 _date = _new_date;
251 break;
252 case _GPS_SENTENCE_GPGGA:
253 _time = _new_time;
254 break;
255 }
256 }
257 }
258 }
259
260#ifndef _GPS_NO_STATS
261 else
262 ++_failed_checksum;
263#endif
264 return false;
265 }
266
267 // the first term determines the sentence type
268 if (_term_number == 0)
269 {
270 if (!gpsstrcmp(_term, _GPRMC_TERM))
271 _sentence_type = _GPS_SENTENCE_GPRMC;
272 else if (!gpsstrcmp(_term, _GPGGA_TERM))
273 _sentence_type = _GPS_SENTENCE_GPGGA;
274 else
275 _sentence_type = _GPS_SENTENCE_OTHER;
276 return false;
277 }
278
279 if (_sentence_type != _GPS_SENTENCE_OTHER && _term[0])
280 switch(COMBINE(_sentence_type, _term_number))
281 {
282 case COMBINE(_GPS_SENTENCE_GPRMC, 1): // Time in both sentences
283 case COMBINE(_GPS_SENTENCE_GPGGA, 1):
284 _new_time = parse_decimal();
285 _new_time_fix = millis();
286 break;
287 case COMBINE(_GPS_SENTENCE_GPRMC, 2): // GPRMC validity
288 _gps_data_good = _term[0] == 'A';
289 break;
290 case COMBINE(_GPS_SENTENCE_GPRMC, 3): // Latitude
291 case COMBINE(_GPS_SENTENCE_GPGGA, 2):
292 if (_raw_enabled) {
293 _new_latitude = parse_degrees_raw();
294 } else {
295 _new_latitude = parse_degrees();
296 }
297 _new_position_fix = millis();
298 break;
299 case COMBINE(_GPS_SENTENCE_GPRMC, 4): // N/S
300 case COMBINE(_GPS_SENTENCE_GPGGA, 3):
301 if (_term[0] == 'S')
302 _new_latitude = -_new_latitude;
303 break;
304 case COMBINE(_GPS_SENTENCE_GPRMC, 5): // Longitude
305 case COMBINE(_GPS_SENTENCE_GPGGA, 4):
306 if (_raw_enabled) {
307 _new_longitude = parse_degrees_raw();
308 } else {
309 _new_longitude = parse_degrees();
310 }
311 break;
312 case COMBINE(_GPS_SENTENCE_GPRMC, 6): // E/W
313 case COMBINE(_GPS_SENTENCE_GPGGA, 5):
314 if (_term[0] == 'W')
315 _new_longitude = -_new_longitude;
316 break;
317 case COMBINE(_GPS_SENTENCE_GPRMC, 7): // Speed (GPRMC)
318 _new_speed = parse_decimal();
319 break;
320 case COMBINE(_GPS_SENTENCE_GPRMC, 8): // Course (GPRMC)
321 _new_course = parse_decimal();
322 break;
323 case COMBINE(_GPS_SENTENCE_GPRMC, 9): // Date (GPRMC)
324 _new_date = gpsatol(_term);
325 break;
326 case COMBINE(_GPS_SENTENCE_GPGGA, 6): // Fix data (GPGGA)
327 _gps_data_good = _term[0] > '0';
328 break;
329 case COMBINE(_GPS_SENTENCE_GPGGA, 7): // Satellites used (GPGGA)
330 _new_numsats = (unsigned char)atoi(_term);
331 break;
332 case COMBINE(_GPS_SENTENCE_GPGGA, 8): // HDOP
333 _new_hdop = parse_decimal();
334 break;
335 case COMBINE(_GPS_SENTENCE_GPGGA, 9): // Altitude (GPGGA)
336 _new_altitude = parse_decimal();
337 break;
338 }
339
340 return false;
341}

344{
345 long ret = 0;

359float TinyGPS::distance_between (float lat1, float long1, float lat2, float long2)
360{
361 // returns distance in meters between two positions, both specified
362 // as signed decimal-degrees latitude and longitude. Uses great-circle
363 // distance computation for hypothetical sphere of radius 6372795 meters.
364 // Because Earth is no exact sphere, rounding errors may be up to 0.5%.
365 // Courtesy of Maarten Lamers
366 float delta = radians(long1-long2);
367 float sdlong = sinf(delta);
368 float cdlong = cosf(delta);
369 lat1 = radians(lat1);
370 lat2 = radians(lat2);
371 float slat1 = sinf(lat1);
372 float clat1 = cosf(lat1);
373 float slat2 = sinf(lat2);
374 float clat2 = cosf(lat2);
375 delta = (clat1 * slat2) - (slat1 * clat2 * cdlong);
376 delta = sqrtf(delta);
377 delta += sqrtf(clat2 * sdlong);
378 delta = sqrtf(delta);
379 float denom = (slat1 * slat2) + (clat1 * clat2 * cdlong);
380 delta = atan2f(delta, denom);
381 return delta * 6372795;
382}

384float TinyGPS::course_to (float lat1, float long1, float lat2, float long2)
385{
386 // returns course in degrees (North=0, West=270) from position 1 to position 2,
387 // both specified as signed decimal-degrees latitude and longitude.
388 // Because Earth is no exact sphere, calculated course may be off by a tiny fraction.
389 // Courtesy of Maarten Lamers
390 float dlon = radians(long2-long1);
391 lat1 = radians(lat1);
392 lat2 = radians(lat2);
393 float a1 = sinf(dlon) * cosf(lat2);
394 float a2 = sinf(lat1) * cosf(lat2) * cosf(dlon);
395 a2 = cosf(lat1) * sinf(lat2) - a2;
396 a2 = atan2f(a1, a2);
397 if (a2 < 0.0)
398 {
399 a2 += TWO_PI;
400 }
401 return degrees(a2);
402}

405{
406 static const char* directions[] = {"N", "NNE", "NE", "ENE", "E", "ESE", "SE", "SSE", "S", "SSW", "SW", "WSW", "W", "WNW", "NW", "NNW"};

409 return directions[direction % 16];
