150void SdMessageScreen::enter_screen(Controller& controller) {

152 auto local_available = controller.get_data().local_available;
153 switch (sd_status) {
154 case SDInterface::e_sd_config_status_not_ready:
155 error_type = local_available ? SdMessageType::k_no_sd_with_storage : SdMessageType::k_no_sd_no_storage;
156 break;
157 case SDInterface::e_sd_config_status_ok:
158 error_type = local_available ? SdMessageType::k_unknown : SdMessageType::k_new_device;
159 break;
160 case SDInterface::e_sd_config_status_config_different_id:
161 error_type = SdMessageType::k_config_sd_different_id;
162 break;
163 case SDInterface::e_sd_config_status_config_no_content:
164 case SDInterface::e_sd_config_status_no_config_file:
