194void ConfigWebServer::handle_update_uploading() {
195 HTTPUpload& upload = _server.upload();
196 switch(upload.status) {
197 case UPLOAD_FILE_START: {
198 DEBUG_PRINTF("Update: %s\n", upload.filename.c_str());
199 if(Update.begin(UPDATE_SIZE_UNKNOWN)) {
200 DEBUG_PRINTLN("Starting update");
201 } else {
202 DEBUG_PRINTLN("Unable to start update");
203 }
204 break;
205 }
206 case UPLOAD_FILE_WRITE: {
207 DEBUG_PRINTF(".");
208 auto write_size = Update.write(upload.buf, upload.currentSize);
209 if(write_size != upload.currentSize) {
210 DEBUG_PRINTF("Something failed while uploading (wrote %d out of %d)\n", write_size, upload.currentSize);
211 Update.abort();
212 }
213 break;
214 }
215 case UPLOAD_FILE_END: {
216 if(upload.totalSize > 0 && Update.end(true)) { //true to set the size to the current progress
217 DEBUG_PRINTF("\nUpdate Success: %u\n", upload.totalSize);
218 } else {
219 DEBUG_PRINTF("Update Failed...");
220 }
221 break;
222 }
223 case UPLOAD_FILE_ABORTED: {
224 Update.abort();
225 DEBUG_PRINTF("Update aborted...");
226 break;
227 }
228 }
229}
