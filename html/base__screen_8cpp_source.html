24void BaseScreen::drawButton(uint16_t x, const char* text, ButtonState state) {
25 if (strlen(text) > BUTTON_TEXT_MAX_LENGTH) {
26 // Don't render long button text :(
27 return;
28 }
29
30 int border_color = LCD_COLOR_BACKGROUND;
31 int text_color = LCD_COLOR_STALE_INCOMPLETE;
32
33 if (strlen(text)) {
34 switch (state) {
35 case e_button_default:
36 border_color = LCD_COLOR_DEFAULT;
37 break;
38 case e_button_active:
39 border_color = LCD_COLOR_STALE_INCOMPLETE;
40 break;
41 case e_button_disabled:
42 border_color = LCD_COLOR_INACTIVE;
43 text_color = LCD_COLOR_INACTIVE;
44 break;
45 }
46 }
47
48 M5.Lcd.drawRoundRect(x, -5, 90, 20, 4, border_color);
49 M5.Lcd.setTextColor(text_color, LCD_COLOR_BACKGROUND);
50
51 // Center the button text, making sure around the text is whitespace to clear previous texts
52 char blob[BUTTON_TEXT_MAX_LENGTH + 1] = " ";
53 for (size_t i = 0; i < strlen(text); ++i) {
54 blob[i + ((BUTTON_TEXT_MAX_LENGTH - strlen(text)) / 2)] = text[i];
55 }
56 if (strlen(text) % 2 != BUTTON_TEXT_MAX_LENGTH % 2) {
57 blob[BUTTON_TEXT_MAX_LENGTH - 1] = '\0';
58 }
59 M5.Lcd.drawString(blob, (x + 45) - static_cast<uint16_t>(strlen(blob) * 3), 10, &fonts::Font0);
60 M5.Lcd.flush();
61}

171 force_next_render();

173
174 // Button 3 change view
175 if (button3->is_fresh() && button3->get_data().shortPress && items[_menu_index].enabled) {
176
177

179 // Swap screens

181 } else {
182 // Swap pages internally

185 if (_menu_index != _current_page) {
