218void ConfigModeScreen::enter_screen(Controller& controller) {
219 switch (_current_page) {
220 case e_config_page_main:
221 // In case when entering from main menu, always set config menu index correctly
222 _menu_index = e_config_page_main;
223 break;
224 case e_config_page_ap:
225 WiFiWrapper_i.start_ap_server(controller.get_settings().get_device_id(), controller.get_settings().get_ap_password());
226 controller.set_worker_active(k_worker_config_server, true);
227 break;
228 case e_config_page_wifi:
229 WiFiWrapper_i.connect_wifi(controller.get_settings().get_wifi_ssid(), controller.get_settings().get_wifi_password());
230 controller.set_worker_active(k_worker_config_server, true);
231 break;
232 case e_config_page_load_sd_config:
233 if (controller.load_sd_config()) {
234 set_status_message(F(" SD CONFIG LOADED, settings have been updated! "));
235 }
236 else {
237 set_status_message(F(" LOAD CONFIG FAILED, no config or SD-card! "));
238 }
239 _current_page = e_config_page_main; // Back on main page
240 open_menu(true); // Re-enter menu
241 break;
242 case e_config_page_save_config_to_sd:
243 if (controller.write_sd_config()) {
244 set_status_message(F(" CONFIG SAVED, settings have been saved to SD! "));
245 }
246 else {
247 set_status_message(F(" WRITE CONFIG FAILED, no SD-card! "));
248 }
249 _current_page = e_config_page_main; // Back on main page
250 open_menu(true); // Re-enter menu
251 break;
252 default:
253 break;
254 }
