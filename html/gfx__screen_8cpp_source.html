50void GFXScreen::setBrightness(uint8_t lvl, bool overdrive) {
51#ifdef M5_CORE2
52 // The backlight brightness is in steps of 25 in AXP192.cpp
53 // calculation in SetDCVoltage: ( (voltage - 700) / 25 )
54 // 2325 is the minimum "I can just about see a glow in a dark room" level of brightness.
55 // 2800 is the value set by the AXP library as "standard" bright backlight.
56 int v = lvl * 25 + 2300;
57
58 // Clamp to safe values.
59 if (v < 2300) v = 2300;
60 if (overdrive) {
61 if (v > 3200) v = 3200; // maximum of 3.2 volts, 3200 (uint8_t lvl = 36) absolute max!
62 } else {
63 if (v > 2800) v = 2800; // maximum of 2.8 volts, 2800 (uint8_t lvl = 20)
64 }
65
66 // Minimum brightness means turn off the LCD backlight.
67 if (v == 2300) {
68 // LED set to minimum brightness? Turn off.
69
70 M5.Axp.SetDCDC3(false);
71 return;
72 } else {
73 // Ensure backlight is on. (magic name = DCDC3)
74 M5.Axp.SetDCDC3(true);
75 }
76
77 // Set the LCD backlight voltage. (magic number = 2)
78
79 M5.Axp.SetDCVoltage(2, v);
80
81#elif M5_BASIC
82 // M5Stack Basic uses LCD Brightness (0: Off - 255:Full)
83 int v = lvl * 25;
84 M5.Lcd.setBrightness(v);
85#endif
86}
