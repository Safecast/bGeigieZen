8void core2Brightness(uint8_t lvl, bool overdrive = false) {
9 // The backlight brightness is in steps of 25 in AXP192.cpp
10 // calculation in SetDCVoltage: ( (voltage - 700) / 25 )
11 // 2325 is the minimum "I can just about see a glow in a dark room" level of brightness.
12 // 2800 is the value set by the AXP library as "standard" bright backlight.
13 int v = lvl * 25 + 2300;
14
15 // Clamp to safe values.
16 if (v < 2300) v = 2300;
17 if (overdrive) {
18 if (v > 3200) v = 3200; // maximum of 3.2 volts, 3200 (uint8_t lvl = 36) absolute max!
19 } else {
20 if (v > 2800) v = 2800; // maximum of 2.8 volts, 2800 (uint8_t lvl = 20)
21 }
22
23 // Minimum brightness means turn off the LCD backlight.
24 if (v == 2300) {
25 // LED set to minimum brightness? Turn off.
26 M5.Axp.SetDCDC3(false);
27 return;
28 } else {
29 // Ensure backlight is on. (magic name = DCDC3)
30 M5.Axp.SetDCDC3(true);
31 }
32
33 // Set the LCD backlight voltage. (magic number = 2)
34 M5.Axp.SetDCVoltage(2, v);
35}

441void printFloat(float val, bool valid, int len, int prec) {
442 if (!valid) {
443 while (len-- > 1) M5.Lcd.print('*');
444 M5.Lcd.print(' ');
445 } else {
446 M5.Lcd.print(val, prec);
447 int vi = abs((int)val);
448 int flen = prec + (val < 0.0 ? 2 : 1); // . and -
449 flen += vi >= 1000 ? 4 : vi >= 100 ? 3 : vi >= 10 ? 2 : 1;
450 for (int i = flen; i < len; ++i) M5.Lcd.print(' ');
451 }
452 delay(0);
