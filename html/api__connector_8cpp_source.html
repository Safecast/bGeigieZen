39int8_t ApiConnector::handle_produced_work(const worker_map_t& workers) {
40 if (!_config.get_fixed_device_id()) {
41 // Cant even send anything, because there is an invalid id
42 return e_handler_idle;
43 }
44
45 const auto& log_aggregator = workers.worker<LogAggregator>(k_worker_log_aggregator);
46 const auto& log_data = log_aggregator->get_data();
47
48 if (!log_aggregator->is_fresh()) {
49 // No fresh data
50 return e_handler_idle;
51 }
52
53 if (!log_data.valid()) {
54 // data not valid (either gm or gps)
55 return e_handler_idle;
56 }
57
58 if (!time_to_send(log_data.in_fixed_range, log_data.cpm > _config.get_alert_threshold())) {
59 // Not time to send yet
60 return e_handler_idle;
61 }
62
63 if (!WiFi.isConnected() && !activate(true)) {
64 M5_LOGD("Unable to send, lost connection");
65 return e_api_reporter_error_not_connected;
66 }
67
68 if (!reading_to_json(log_data,_payload)) {
69 return e_api_reporter_error_to_json;
70 }
71
72 M5_LOGD("Starting task to send data to API...");
73 return start_task("api_send", 2048 * 4, 3);
74}
