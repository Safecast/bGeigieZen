108 // Analyze each "module" in the QR code (each pixel), and convert it to an
109 // Arduino-compatible bitmap. Note that if we are QRDOUBLE'ing its size for
110 // readability, we need to set two actual pixels horizontally for each pixel
111 // in the source, and we need to duplicate each line.
112 for (uint8_t y = 0; y < qrcode.size; y++) {
113#ifdef QRDOUBLE
114 uint16_t px = 0;
115#else
116 uint8_t px = 0;
117#endif
118 uint8_t *q = p;
119 for (uint8_t x = 0; x < qrcode.size; x++) {
120 if ((x & 0x07) == 0)
121 px = 0;
122#ifdef QRDOUBLE
123 if (qrcode_getModule(&qrcode, x, y))
124 px |= (3 << (2*(7 - (x & 0x07))));
125 *q = (uint8_t) (px >> 8);
126 *(q+1) = (uint8_t) (px & 0xff);
127#else
128 if (qrcode_getModule(&qrcode, x, y))
129 px |= (1 << (7 - (x & 0x07)));
130 *q = px;
131#endif
132 if ((x & 0x07) == 7) {
133 q++;
134#ifdef QRDOUBLE
135 q++;
136#endif
137 }
138 }
139#ifdef QRDOUBLE
140 memcpy(&p[QRPIXELS/8], p, QRPIXELS/8);
141 p += QRPIXELS/8;
142#endif
143 p += QRPIXELS/8;
144 }
145 }
146 _free(qrcodeBytes);
