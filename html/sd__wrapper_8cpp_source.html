516void SDInterface::delete_directory(const char* dirPath) {
517 File dir = SD.open(dirPath);
518 if (!dir) {
519 M5_LOGD("Directory does not exist or could not be opened.");
520 return;
521 }
522
523 if (!dir.isDirectory()) {
524 M5_LOGD("The specified path is not a directory.");
525 dir.close();
526 return;
527 }
528
529 // Iterate through all files in the directory
530 while (true) {
531 File entry = dir.openNextFile();
532 if (!entry) {
533 // No more files in the directory
534 break;
535 }
536
537 // Construct the full path
538 String fullPath = String(dirPath) + "/" + entry.name();
539
540 if (entry.isDirectory()) {
541 // Recursively delete subdirectories
542 M5_LOGD("Recursively deleting directory: %s", fullPath.c_str());
543 delete_directory(fullPath.c_str());
544
545 // Attempt to remove the directory itself after its contents are deleted
546 if (!SD.rmdir(fullPath.c_str())) {
