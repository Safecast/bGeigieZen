144void SdMessageScreen::enter_screen(Controller& controller) {
145 auto sd_status = controller.get_data().sd_card_status;
146 auto local_available = controller.get_data().local_available;
147 switch (sd_status) {
148 case SDInterface::e_sd_config_status_not_ready:
149 error_type = local_available ? SdMessageType::k_no_sd_with_storage : SdMessageType::k_no_sd_no_storage;
150 break;
151 case SDInterface::e_sd_config_status_ok:
152 error_type = local_available ? SdMessageType::k_unknown : SdMessageType::k_config_sd_different_id;
153 break;
154 case SDInterface::e_sd_config_status_config_different_id:
155 error_type = SdMessageType::k_config_sd_different_id;
156 break;
157 case SDInterface::e_sd_config_status_config_no_content:
158 case SDInterface::e_sd_config_status_no_config_file:
159 error_type = local_available ? SdMessageType::k_empty_sd_with_storage : SdMessageType::k_empty_sd_no_storage;
160 break;
161 }
