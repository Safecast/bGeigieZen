44void GFXScreen::setBrightness(uint8_t lvl, bool overdrive) {
45#ifdef M5_CORE2
46 // The backlight brightness is in steps of 25 in AXP192.cpp
47 // calculation in SetDCVoltage: ( (voltage - 700) / 25 )
48 // 2325 is the minimum "I can just about see a glow in a dark room" level of brightness.
49 // 2800 is the value set by the AXP library as "standard" bright backlight.
50 int v = lvl * 25 + 2300;
51
52 // Clamp to safe values.
53 if (v < 2300) v = 2300;
54 if (overdrive) {
55 if (v > 3200) v = 3200; // maximum of 3.2 volts, 3200 (uint8_t lvl = 36) absolute max!
56 }
57 else {
58 if (v > 2800) v = 2800; // maximum of 2.8 volts, 2800 (uint8_t lvl = 20)
59 }
60
61 // Minimum brightness means turn off the LCD backlight.
62 if (v == 2300) {
63 // LED set to minimum brightness? Turn off.
64
65 M5.Axp.SetDCDC3(false);
66 return;
67 }
68 else {
69 // Ensure backlight is on. (magic name = DCDC3)
70 M5.Axp.SetDCDC3(true);
71 }
72
73 // Set the LCD backlight voltage. (magic number = 2)
74
75 M5.Axp.SetDCVoltage(2, v);
76
77#elif M5_BASIC
78 // M5Stack Basic uses LCD Brightness (0: Off - 255:Full)
79 int v = lvl * 25;
80 M5.Lcd.setBrightness(v);
81#endif
82}
