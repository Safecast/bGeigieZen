51int8_t ApiConnector::handle_produced_work(const worker_map_t& workers) {
52 const auto& data_worker = workers.worker<LogAggregator>(k_worker_log_aggregator);
53
54 if (!data_worker || !data_worker->is_fresh()) {
55 // No fresh data
56 return _current_default_response;
57 }
58

63 }
64 return _current_default_response;
65 }
66
67 const auto& line = data_worker->get_data();
68

70 const unsigned long time_at_send = millis();
71 _current_default_response = send_reading(line);
72
73 if (_current_default_response == e_api_reporter_send_success) {
74 _last_post = time_at_send;
75 }
76 }
